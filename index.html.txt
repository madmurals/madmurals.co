    <script type="module">
        const form = document.getElementById('contact-form');
        const statusMessage = document.getElementById('status-message');
        
        form.addEventListener('submit', async function(event) {
            event.preventDefault();

            const formData = new FormData(form);
            const data = Object.fromEntries(formData.entries());

            statusMessage.innerHTML = '<p class="text-gray-500">Sending...</p>';
            
            try {
                const response = await fetch(form.action, {
                    method: 'POST',
                    headers: {
                        'Accept': 'application/json'
                    },
                    body: JSON.stringify(data)
                });

                if (response.ok) {
                    statusMessage.innerHTML = '<p class="text-green-600 font-semibold">Thank you for your submission! We will get back to you soon.</p>';
                    form.reset();
                } else {
                    statusMessage.innerHTML = '<p class="text-red-600 font-semibold">Something went wrong. Please try again later.</p>';
                    console.error('Form submission failed:', response.statusText);
                }
            } catch (error) {
                statusMessage.innerHTML = '<p class="text-red-600 font-semibold">Something went wrong. Please try again later.</p>';
                console.error('Fetch error:', error);
            }
        });

        // Get the modal and the image element inside it
        const modal = document.getElementById("myModal");
        const modalImg = document.getElementById("modalImage");

        // Function to open the modal
        window.openModal = function(src) {
            modal.style.display = "flex";
            modalImg.src = src;
        }

        // Function to close the modal
        window.closeModal = function() {
            modal.style.display = "none";
        }

        // Close the modal if the user clicks anywhere outside of the image
        window.onclick = function(event) {
            if (event.target == modal) {
                modal.style.display = "none";
            }
        }
    </script>
